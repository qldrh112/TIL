```python
def charge_money(won):
    money = [50000, 10000, 5000, 1000, 500, 100, 50, 10]
    money_count = [0] * 8

    while won > 0:
        for i in range(8):
            while won >= money[i]:
                won -= money[i]
                money_count[i] += 1

    return money_count

T = int(input())
for t in range(T):
    won = int(input())
    print(f'#{t+1}') 
    print(*charge_money(won))
```

```python
def min_bill_charge(won):
    bill = [50000, 10000, 5000, 1000, 500, 100, 50, 10]
    charges = [0] * 8
    for i in range(8):
        charges[i] = won // bill[i]
        won -= charges[i] * bill[i]
        
    return charges

T = int(input())
for t in range(T):
    won = int(input())
    print(f'#{t+1}')
    print(*min_bill_charge(won))
```
